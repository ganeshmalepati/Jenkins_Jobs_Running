pipeline {
    agent any
    stages {
        stage('Clone Repository') {
            steps {
                git 'https://github.com/ganeshmalepati/Random.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    docker.build('Dockerfile')
                }
            }
        }
        stage('Run Docker Image') {
            steps {
                script {
                    docker.image('Dockerfile').inside {
                        sh 'python rand.py'
                    }
                }
            }
        }
    }
}
